📦 Python SDK

The A2Apex Python SDK provides a simple, Pythonic interface for validating Agent Cards and testing A2A protocol compliance. Perfect for CI/CD integration and automated testing.

📋 Requirements

Python 3.10+ · Dependencies: httpx, pydantic≥2.0

Installation

pip install a2apex

For rich terminal output with colored tables:

pip install a2apex[rich]

Quick Start

Validate an Agent Card

from a2apex import A2ApexClient

client = A2ApexClient()

# Validate from URL (fetches /.well-known/agent-card.json)
report = client.validate_card("https://agent.example.com")

print(f"Score: {report.score}/100")
print(f"Valid: {report.is_valid}")
print(f"Errors: {report.error_count}")
print(f"Warnings: {report.warning_count}")

# Show errors
for error in report.errors:
    print(f"  ✗ {error.field}: {error.message}")
    if error.suggestion:
        print(f"    💡 {error.suggestion}")

Run Full Test Suite

from a2apex import A2ApexClient

client = A2ApexClient()
results = client.test_agent("https://agent.example.com")

print(f"Score: {results.score}/100")
print(f"Passed: {results.passed}/{results.total_tests}")

for test in results:
    status = "✅" if test.passed else "❌"
    print(f"{status} {test.name}: {test.message}")

Validate a Dict (No HTTP)

from a2apex import validate_agent_card

card = {
    "name": "My Agent",
    "url": "https://api.example.com/a2a",
    "version": "1.0.0",
    "capabilities": {"streaming": True},
    "skills": [
        {"id": "chat", "name": "Chat", "description": "General conversation"}
    ]
}

report = validate_agent_card(card)
print(report)  # ✓ Valid (Score: 75/100) — 0 errors, 5 warnings

A2ApexClient

The main entry point for the SDK.

Constructor

client = A2ApexClient(
    timeout=30.0,        # Request timeout in seconds
    auth_header=None,    # Default Authorization header for all requests
)

Methods

MethodReturnsDescription
validate_card(url_or_dict) ValidationReport Validate an Agent Card (sync)
avalidate_card(url_or_dict) ValidationReport Validate an Agent Card (async)
test_agent(url) TestReport Run full test suite (sync)
atest_agent(url) TestReport Run full test suite (async)

ValidationReport

Returned by validate_card():

report.is_valid      # bool - No errors
report.score         # float - 0-100 compliance score
report.error_count   # int - Number of errors
report.warning_count # int - Number of warnings
report.errors        # list[ValidationIssue] - Must fix
report.warnings      # list[ValidationIssue] - Should fix
report.info          # list[ValidationIssue] - Suggestions

ValidationIssue

issue.field          # str - Field path (e.g., "skills[0].id")
issue.message        # str - Error/warning message
issue.severity       # Severity - ERROR, WARNING, or INFO
issue.suggestion     # str | None - How to fix
issue.spec_reference # str | None - Reference to spec section

TestReport

Returned by test_agent():

results.score           # float - 0-100 compliance score
results.passed          # int - Passed tests
results.failed          # int - Failed tests
results.warnings        # int - Tests with warnings
results.skipped         # int - Skipped tests
results.total_tests     # int - Total tests run
results.results         # list[TestResult] - All results

# Iterate over results
for test in results:
    print(test.name, test.passed, test.message)

TestResult

test.name     # str - Test identifier
test.passed   # bool - Test passed
test.status   # str - "passed", "failed", "warning", "skipped"
test.message  # str - Result description
test.details  # dict | None - Additional details
test.duration # float | None - Test duration in ms

State Machine Validation

Validate A2A task state transitions:

from a2apex import validate_transitions, is_terminal_state, get_valid_next_states

# Check if a sequence of state transitions is valid
result = validate_transitions(["submitted", "working", "completed"])
print(f"Valid: {result.is_valid}")  # True

# Invalid transition (can't go from completed to working)
result = validate_transitions(["submitted", "working", "completed", "working"])
print(f"Valid: {result.is_valid}")  # False

for v in result.violations:
    print(f"  ✗ {v.from_state} → {v.to_state}: {v.reason}")

# Check terminal states
print(is_terminal_state("completed"))  # True
print(is_terminal_state("working"))    # False

# Get valid next states
next_states = get_valid_next_states("working")
# ['input-required', 'auth-required', 'completed', 'failed', 'canceled']

Async Support

All methods have async versions for use with asyncio:

import asyncio
from a2apex import A2ApexClient

async def main():
    client = A2ApexClient()
    
    # Async validation
    report = await client.avalidate_card("https://agent.example.com")
    print(f"Score: {report.score}/100")
    
    # Async testing
    results = await client.atest_agent("https://agent.example.com")
    print(f"Passed: {results.passed}/{results.total_tests}")

asyncio.run(main())

CI/CD Integration

pytest Example

# test_a2a_compliance.py
from a2apex import A2ApexClient
import pytest

@pytest.fixture
def client():
    return A2ApexClient(timeout=60)

def test_agent_card_valid(client):
    """Agent Card passes validation."""
    report = client.validate_card("https://your-agent.example.com")
    assert report.is_valid, f"Errors: {[e.message for e in report.errors]}"

def test_compliance_score(client):
    """Agent achieves minimum compliance score."""
    results = client.test_agent("https://your-agent.example.com")
    assert results.score >= 80, f"Score {results.score} below minimum 80"

def test_critical_endpoints(client):
    """Critical A2A endpoints work correctly."""
    results = client.test_agent("https://your-agent.example.com")
    
    critical = ["agent_card_fetch", "message_send"]
    for test in results:
        if test.name in critical:
            assert test.passed, f"Critical test failed: {test.name} - {test.message}"

Export Reports

Generate JSON or HTML reports:

from a2apex import A2ApexClient, export_report

client = A2ApexClient()
results = client.test_agent("https://agent.example.com")

# Export to JSON
export_report(results, "report.json")

# Export to HTML
export_report(results, "report.html")

# Get as dict
report_dict = results.to_dict()

Pydantic Models

A2Apex includes complete Pydantic v2 models for all A2A types:

from a2apex import (
    AgentCard,
    Task,
    Message,
    TextPart,
    DataPart,
    FilePart,
    Artifact,
    create_text_message,
)

# Parse an Agent Card
card = AgentCard.model_validate(card_dict)
print(card.name)
print(card.capabilities.streaming)

# Create a message
msg = create_text_message("Hello!", role="user")

# Work with task data
task = Task.model_validate(task_dict)
print(task.status.state)
for artifact in task.artifacts:
    print(artifact.name)

Error Handling

from a2apex import A2ApexClient
from a2apex.exceptions import (
    A2ApexError,
    ValidationError,
    ConnectionError,
    TimeoutError,
)

client = A2ApexClient()

try:
    report = client.validate_card("https://agent.example.com")
except ConnectionError as e:
    print(f"Could not connect: {e}")
except TimeoutError as e:
    print(f"Request timed out: {e}")
except ValidationError as e:
    print(f"Invalid response: {e}")
except A2ApexError as e:
    print(f"A2Apex error: {e}")
